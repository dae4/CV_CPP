cmake_minimum_required(VERSION 3.10)
project(MotionDetector)

set(CMAKE_CXX_STANDARD 14)

# OpenCV 패키지 찾기
find_package(OpenCV REQUIRED)

# 컴파일할 소스 파일들 지정
add_executable(MotionDetector 
    main.cpp
    ImageProcessor.cpp
    DisplayManager.cpp
)

# OpenCV 라이브러리 연결
target_link_libraries(MotionDetector ${OpenCV_LIBS})
